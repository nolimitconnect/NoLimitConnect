=== checkout ===
git clone https://gitlab.com/nolimitconnectapps/nolimitapp.git
cd nolimitapp

=== allow flatpak submodules (required for GL/glu lib) ===
git submodule add https://github.com/flathub/shared-modules.git

=== build nolimitconnect ===
# just build
flatpak-builder build-dir com.nolimitconnect.NoLimitConnect.yml --force-clean

# build and install so is ready to run
flatpak-builder --user --install --force-clean build-dir com.nolimitconnect.NoLimitConnect.yml

# maybe needed?
# flatpak-builder --repo=myrepo --force-clean build-dir com.nolimitconnect.NoLimitConnect.yml
# flatpak --user remote-add --no-gpg-verify myrepo myrepo

=== run nolimitconnect ===
flatpak run com.nolimitconnect.NoLimitConnect

=== list flatpak apps ===
flatpak list --app

=== show NLC info ===
flatpak info --show-location com.nolimitconnect.NoLimitConnect

=== uninstall nolimitconnect ===
flatpak uninstall --delete-data com.nolimitconnect.NoLimitConnect

=== build nolimitconnect notes ===
NOTE: to use local sources change sources of com.nolimitconnect.NoLimitConnect.yml to
   sources:
      - type: dir
        path: ./
