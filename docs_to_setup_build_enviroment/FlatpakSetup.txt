
=== to setup flatpak build enviroment ===
sudo apt install flatpak
flatpak remote-add --if-not-exists flathub https://dl.flathub.org/repo/flathub.flatpakrepo
reboot

=== install flatpak builder ===
sudo apt install flatpak-builder

flatpak install flathub org.kde.Platform
// and pick version 6.7
// and run 
flatpak install flathub org.kde.Sdk
// and pick version 6.7

// this command can be used to list available versions
flatpak -vv remote-ls flathub

=== allow flatpak submodules (required for GL/glu lib) ===
// nolimitconnect app checkout
git clone https://gitlab.com/nolimitconnectapps/nolimitapp.git
// after checkout
cd nolimitapp
git submodule add https://github.com/flathub/shared-modules.git

see CompileNoLimitConnectUsingFlatpak.txt for build instructions
